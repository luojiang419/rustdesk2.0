修改前：

修改后：
```rust
pub struct ServerProfileStatus {
    pub id: String,
    pub id_server: String,
    pub relay_server: String,
    pub latency_ms: i64,
    pub error: String,
    pub relay_latency_ms: i64,
```

修改前：
```rust
let start = Instant::now();
let server = check_port(&profile.id_server, RENDEZVOUS_PORT);
match connect_tcp(server, CONNECT_TIMEOUT).await {
    Ok(_) => ServerProfileStatus {
        id: profile.id,
        id_server: profile.id_server,
        latency_ms: start.elapsed().as_millis() as i64,
        error: String::new(),
    },
    Err(err) => ServerProfileStatus {
        id: profile.id,
        id_server: profile.id_server,
        latency_ms: -1,
        error: err.to_string(),
    },
}
```

修改后：
```rust
let id = profile.id;
let id_server = profile.id_server;
let relay_server = profile.relay_server;
let ((latency_ms, error), (relay_latency_ms, relay_error)) = join(
    test_endpoint_latency(&id_server, RENDEZVOUS_PORT),
    test_endpoint_latency(&relay_server, RELAY_PORT),
)
.await;
ServerProfileStatus {
    id,
    id_server,
    relay_server,
    latency_ms,
    error,
    relay_latency_ms,
    relay_error,
}
```
